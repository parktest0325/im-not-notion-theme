Sprite Batching

For 2D games, the main optimization is sprite batching — combining multiple draw calls into one:

struct SpriteBatch {
    vertices: Vec<Vertex>,
    indices: Vec<u32>,
    texture: TextureId,
}

impl SpriteBatch {
    fn draw_sprite(&mut self, pos: Vec2, size: Vec2, uv: Rect) {
        let base = self.vertices.len() as u32;
        self.vertices.extend_from_slice(&[
            Vertex { pos: pos, uv: uv.top_left() },
            Vertex { pos: pos + vec2(size.x, 0.0), uv: uv.top_right() },
            Vertex { pos: pos + size, uv: uv.bottom_right() },
            Vertex { pos: pos + vec2(0.0, size.y), uv: uv.bottom_left() },
        ]);
        self.indices.extend_from_slice(&[
            base, base + 1, base + 2,
            base, base + 2, base + 3,
        ]);
    }
}

Results

With batching, we can render 10,000+ sprites at 60fps on integrated graphics.
